#### 1.4 创建振动反馈工具【P1】

**文件**：`src/utils/haptic.ts`（新建）

**内容**：
```ts
/**
 * 振动反馈工具（mp-weixin 真实振动，H5 静默失败）
 * 封装 uni.vibrateShort，统一调用入口
 */

/** 轻量振动（按钮点击） */
export function lightHaptic(): void {
  try {
    uni.vibrateShort({ type: 'light' });
  } catch {
    // H5 / 不支持振动的环境静默失败
  }
}

/** 中等振动（卡片滑动） */
export function mediumHaptic(): void {
  try {
    uni.vibrateShort({ type: 'medium' });
  } catch {
    // 静默失败
  }
}

/** 重振动（匹配成功） */
export function heavyHaptic(): void {
  try {
    uni.vibrateShort({ type: 'heavy' });
  } catch {
    // 静默失败
  }
}

/** 成功振动模式（签到成功、匹配成功） */
export function successHaptic(): void {
  try {
    uni.vibrateShort({ type: 'medium' });
    setTimeout(() => uni.vibrateShort({ type: 'light' }), 100);
  } catch {
    // 静默失败
  }
}
```
